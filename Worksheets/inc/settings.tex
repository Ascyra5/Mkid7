%% =================================================================================================================
%%  Definition Section Style
%% =================================================================================================================

\titleformat{\section}
  [block]                                               % Layout: block style
  {\normalfont\sffamily\large\bfseries}                 % Style: normal font, sansserif, large size, bold
  {\hspace*{-0.5cm}\makebox[0pt][r]{\textcolor{\mainc!75!black}{\thesection}}}        % Place the section number on the left margin
  {0.5cm}                                               % Space between the number and the title
  {%
    \textcolor{\mainc!75!black}                          % Shift the section title to align correctly
  }
  [\vspace{-0.325cm}\textcolor{lightgray}{\rule{\linewidth}{1pt}}\vspace{0.5cm}] % Line below the section
%% =================================================================================================================



%% =================================================================================================================
%%  Definition Subsection Style
%% =================================================================================================================

\renewcommand{\thesubsection}{\alph{subsection})}           % Ändert die Nummerierung in section.a, section.b, ...

\titleformat{\subsection}
  [block]                                                   % Layout: block style
  {\normalfont\sffamily\normalsize\bfseries}                  % Style: normal font, sans serif, normal size, bold
  {\hspace*{-0.5cm}\makebox[0pt][r]{\textcolor{\mainc!50!black}{\thesubsection}}} % Platziert die Nummer links
  {0.5cm}                                                     % Abstand zwischen Nummer und Titel
  {%
    \textcolor{\mainc!50!black}                               % Stil des Titels
  }
  [\vspace{-0.325cm}\textcolor{lightgray}{\rule{\linewidth}{0.5pt}}\vspace{0.5cm}] % Linie unterhalb der Überschrift

\titleformat{\subsubsection}
  [block]                                                   % Layout: block style
  {\normalfont\sffamily\normalsize\itshape}                 % Style: normal font, sans serif, normal size, italic (z.B. zur Abgrenzung)
